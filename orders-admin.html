import { db } from "./firebase.js"; import { collection, getDocs } from "https://www.gstatic.com/firebasejs/12.16.0/firebase-firestore.js"; async function loadOrders(){ const snapshot = await getDocs(collection(db,"orders")); const table = document.getElementById("ordersTable"); table.innerHTML = ""; if(snapshot.empty){ table.innerHTML = ` No orders found. `; return; } snapshot.forEach((order)=>{ const data = order.data(); let status = ""; if(data.status==="Completed"){ status = "🟢 Completed"; }else if(data.status==="Expired"){ status = "🔴 Expired"; }else{ status = "🟡 Active"; } let date = ""; if(data.createdAt){ date = data.createdAt.toDate().toLocaleString(); } table.innerHTML += ` ${data.email} ${data.country} ${data.service} ${data.number} ₦${Number(data.price).toLocaleString()} ${data.smsCode || "Waiting..."} ${status} ${date} `; }); } loadOrders();
